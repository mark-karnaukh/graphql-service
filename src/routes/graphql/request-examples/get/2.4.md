### Get user by id with his posts, profile, memberType.

```js
{
    "id": "c9ae803b-3464-441f-b456-3e21beeaed2d"  // Replace with your ID 
}
```

```graphql
query($id: String) {
    user(id: $id) {
        id
        firstName
        lastName
        email
        subscribedToUserIds
        posts {
            id
            title,
            content
            userId
        }
        profile {
            id
            avatar
            sex
            birthday
            country
            street
            city
            memberTypeId
            userId 
        }
        memberType {
            id
            discount
            monthPostsLimit
        }
    }
}

```
